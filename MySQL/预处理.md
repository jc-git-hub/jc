# 预处理

```sql
# 预处理
# 为防止SQL注入
prepare 预处理名  from ‘select * from 表名 where id>?’
# 创建预处理
PREPARE pre1 FROM 'SELECT * FROM person WHERE id>?';
# 设置变量
SELECT @a:=1;
SET @a=5;
# 运行预处理
EXECUTE pre1 USING @a;
# 删除预处理
DROP PREPARE pre1;
```

# 导入导出数据(备份数据)

```sql
# 导入导出
# 表结构占据很大的空间 原则上备份一次表结构就好
# 剩下只需要导入导出数据
# 数据备份
INFILE 导入
outfile 导出

# 查看保存文件位置
show VARIABLES LIKE '%secure_file_priv%';
# 修改配置文件
# my.ini


# 从表 导出数据
SELECT * FROM person into OUTFILE 'c:/mysql_test/test1.txt';

# 清空原表
TRUNCATE TABLE person

SELECT * FROM person
# 导入数据 到表
LOAD data INFILE 'c:/mysql_test/test1.txt' INTO TABLE person;

SELECT * FROM person
```
